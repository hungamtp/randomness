const text%number = document.getElementById("text%number");
text%number.innerHTML =  number[0];
function textF%number ( ){
  let a =Math.random() < 0.5;
  let b =Math.random() > 0.5;
  if(a){
    if(b){
      let valueXIncrease = text%number.cx["animVal"].value +1;
      text%number.style.cx = valueXIncrease;
      text%number.setAttribute("cx" , valueXIncrease);
      text%number.innerHTML = number[0];
    }else{
      let valueXIncrease = text%number.cx["animVal"].value -1;
      text%number.style.cx = valueXIncrease;
      text%number.setAttribute("cx" , valueXIncrease);
      text%number.innerHTML = number[1];
    }
  }else{
    if(b){
      let valueXIncrease = text%number.cy["animVal"].value +1;
      text%number.style.cy = valueXIncrease;
      text%number.setAttribute("cy" , valueXIncrease);
      text%number.innerHTML = number[2];
    }else{
      let valueXIncrease = text%number.cy["animVal"].value -1;
      text%number.style.cy = valueXIncrease;
      text%number.setAttribute("cy" , valueXIncrease);
      text%number.innerHTML =number[3];
    }
  }
}
setInterval(textF%number , 0.0001)
